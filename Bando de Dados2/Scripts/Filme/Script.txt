
CREATE TABLE Filme ( 
       CodFILME       Integer  NOT NULL,
       Titulo          Varchar(25),
       Ano             integer,
       Duracao         integer,
       CodCATEG       integer,
       CodEst         integer);

CREATE TABLE Artista ( 
       CodART      integer  NOT NULL,
       NomeART    Varchar(25),
       Cidade          Varchar(20),
       Pais            Varchar(20),
       DataNasc       Date);

CREATE TABLE Estudio ( 
       CodEst     integer  NOT NULL,
       NomeEst    Varchar(25));

CREATE TABLE Categoria ( 
       CodCATEG       integer  NOT NULL,
       DescCATEG VARCHAR(25));

CREATE TABLE Personagem ( 
       CodART     integer  NOT NULL,
       CodFILME       integer  NOT NULL,
       NomePers  VARCHAR(25),
       Cache           numeric(15,2));
